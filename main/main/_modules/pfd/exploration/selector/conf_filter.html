    def check(
        self,
        conf: List[Atoms],
    ) -> List[Atoms]:
        selected_idx = np.arange(len(conf))
        for ff in self._filters:
            fsel = np.where([ff.check(conf[ii]) for ii in range(len(conf))])[0]
            selected_idx = np.intersect1d(selected_idx, fsel)
        return [conf[ii] for ii in selected_idx]
